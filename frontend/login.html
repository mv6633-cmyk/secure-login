function login() { const username = document.getElementById("username").value; const password = document.getElementById("password").value; fetch("http://localhost:5000/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ username: username, password: password }) }) .then(response => response.json()) .then(data => { document.getElementById("result").innerText = data.message; if (data.message === "Login successful") { document.getElementById("result").style.color = "green"; } else { document.getElementById("result").style.color = "red"; } }) .catch(error => { document.getElementById("result").innerText = "Server error"; document.getElementById("result").style.color = "red"; }); }
